l = int(B/Ca + B/Cb )# total no of element we will be working on
gamma = 0.150991; beta= 0.332161
qc = QuantumCircuit(l, l)
qc.h(range(l))
qc.barrier()
for i in range (l):    
    qc.x(i)
    qc.u1(4*gamma*(((B^2))), i)
    qc.x(i)
    qc.u1(4*gamma*((B^2)), i)

qc.barrier()
H = [Ha,Ha,Hb,Hb,Hb,Hb]
C = [Ca,Ca,Cb,Cb,Cb,Cb]
for i in range (l):
    qc.u1(-gamma*(H[i]+8*C[i]*B-4*(C[i]^2)),i)
qc.barrier()
for i in range(l):
    for j in range(l):
        if i>j:
            qc.cu1(8*gamma*C[i]*C[j],i,j)

qc.barrier()
qc.rx(2*beta, range(l))
qc.barrier()
qc.measure(range(l),range(l))
qc.draw(output='mpl')
